
## 反转单链表

## Python
``` python
class ListNode:
    def __init__(self, val=0, next=None):
        self. Val = val
        self. Next = next

class Solution:
    def reverseList (self, head: Optional[ListNode]) -> Optional[ListNode]:
        pre = nxt = None
        cur = head
        while cur:
            nxt = cur. Next
            cur. Next = pre
        
            pre = cur
            cur = nxt
            
        return pre 


```


## C++
``` c++
#include <vector>

class ListNode {
public:
	int val;
	ListNode* next;
};

class Solution {
	void reverseLsit(ListNode* head) {
		ListNode *cur = head, *pre, *nxt;
		while (cur) {
			nxt = cur->next;
			cur->next = pre;
			
			pre = cur;
			cur = nxt;
		}
		
		return pre;
	}
};
```
